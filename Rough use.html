Jquery shorthands

click to hide()

        <h4 id="clicks">click to hide</h4>
        <button id="click">hide</button>
        <button id="show">show</button>

        $(document).ready(function(){
            $("#click").click(function(){
                $("#clicks").hide()
            });
            $("#show").click(function(){
                $("#clicks").show();
            });
          })

slidetoggle() Up/Down

        <h4 id="slidetoggle">slidetoggle() Up/Down</h4>
        <input id="slideToggle" type="button" value="h2">

        $(document).ready(function(){
            $("#slideToggle").click(function(){
                $("#slidetoggle").slideToggle()
            })
          })
